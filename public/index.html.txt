<script>
    const API_BASE = '';   // Leave empty for Render (same domain)

    async function apiRequest(url, method = 'GET', body = null) {
        const options = {
            method: method,
            headers: { 'Content-Type': 'application/json' }
        };
        if (body) options.body = JSON.stringify(body);

        try {
            const res = await fetch(API_BASE + url, options);
            return await res.json();
        } catch (err) {
            console.error(err);
            return { success: false, message: 'خطا در ارتباط با سرور' };
        }
    }

    // ==================== LOGIN ====================
    async function handleLogin() {
        const username = document.getElementById('username').value.trim();
        const password = document.getElementById('password').value.trim();

        const result = await apiRequest('/api/login', 'POST', { username, password });

        if (result.success) {
            document.getElementById('loginPage').style.display = 'none';
            document.getElementById('app').style.display = 'block';
            loadOrphans();   // Load initial data
        } else {
            alert('❌ نام کاربری یا رمز عبور اشتباه است\n\nadmin / 1234');
        }
    }

    // ==================== ORPHANS ====================
    async function loadOrphans() {
        const data = await apiRequest('/api/orphans');
        const container = document.getElementById('orphans-list');
        if (!data || data.length === 0) {
            container.innerHTML = '<p>هیچ یتیمی یافت نشد</p>';
            return;
        }
        let html = '<table border="1" style="width:100%;border-collapse:collapse"><tr><th>نام</th><th>نام پدر</th><th>سن</th><th>صنف</th></tr>';
        data.forEach(o => {
            html += `<tr><td>${o.name}</td><td>${o.fatherName||'-'}</td><td>${o.age||'-'}</td><td>${o.grade||'-'}</td></tr>`;
        });
        html += '</table>';
        container.innerHTML = html;
    }

    // ==================== SECTION SWITCH ====================
    function showSection(sectionId) {
        document.querySelectorAll('.section').forEach(s => s.classList.remove('active'));
        document.getElementById(sectionId).classList.add('active');
    }

    // ==================== FINANCE MODAL (Example) ====================
    function openFinanceModal() {
        alert('Modal ثبت تراکنش مالی باز شد (بعداً کامل می‌شود)');
        // You can expand this later
    }

    // Allow Enter key on login
    document.addEventListener('keypress', (e) => {
        if (e.key === 'Enter') handleLogin();
    });

    // Initial check
    console.log('%cSystem loaded successfully - API Mode', 'color:#c9a84c;font-weight:bold');
</script>
